# SyncB

#+begin_src bash
# Instalar dependencias
julia --project=@. -e 'using Pkg; Pkg.instantiate()'

# Ejecutar
julia --project=@. -e 'using SyncB; SyncB.main()' --subir --dry-run

# O usar el wrapper
chmod +x bin/syncb
./bin/syncb --subir --verbose
#+end_src
